const express = require('express');
const multer  = require('multer');
const path = require('path');
const fs = require('fs'); // Tambahkan modul File System
const app = express();
const port = 3000;

// Path folder tujuan
const uploadDir = 'E:\\Project\\Labs\\autoprint\\hasil';

// CEK & BUAT FOLDER OTOMATIS: 
// Jika folder belum ada, Node.js akan membuatnya sekarang juga
if (!fs.existsSync(uploadDir)) {
  fs.mkdirSync(uploadDir, { recursive: true });
  console.log('Folder "hasil" belum ada, sistem membuatnya otomatis.');
}

// Konfigurasi tempat penyimpanan file
const storage = multer.diskStorage({
  destination: function (req, file, cb) {
    cb(null, uploadDir); 
  },
  filename: function (req, file, cb) {
    // Membuat nama file unik berdasarkan waktu saat ini
    cb(null, 'print_' + Date.now() + path.extname(file.originalname));
  }
});

const upload = multer({ storage: storage });

// Endpoint untuk menerima upload dari iPad
app.post('/upload', upload.single('image'), (req, res) => {
  if (!req.file) {
    return res.status(400).send({ status: 'error', message: 'Tidak ada file yang dikirim' });
  }

  console.log('File berhasil masuk ke folder:', req.file.path);
  res.send({ status: 'success', message: 'File masuk ke folder hasil' });
});

app.listen(port, '0.0.0.0', () => {
  console.log(`Server siap menerima gambar di port ${port}`);
});
