const express = require('express');
const multer  = require('multer');
const path = require('path');
const { exec } = require('child_process');
const app = express();
const port = 3000;

const storage = multer.diskStorage({
  destination: function (req, file, cb) {
    cb(null, 'C:\\SistemAutoPrint\\HasilFoto'); 
  },
  filename: function (req, file, cb) {
    cb(null, 'print_' + Date.now() + path.extname(file.originalname));
  }
});

const upload = multer({ storage: storage });

app.post('/upload', upload.single('image'), (req, res) => {
  const filePath = req.file.path;
  console.log('Gambar diterima:', filePath);
  
  // Perintah otomatis print menggunakan Windows Paint (tanpa buka aplikasi)
  exec(`mspaint /pt "${filePath}"`, (err) => {
    if (err) {
      console.error('Gagal print:', err);
    } else {
      console.log('Perintah print berhasil dikirim ke printer default!');
    }
  });
  
  res.send({ status: 'success', message: 'Terkirim dan sedang diprint' });
});

app.listen(port, '0.0.0.0', () => {
  console.log(`Server siap menerima gambar di port ${port}`);
});
